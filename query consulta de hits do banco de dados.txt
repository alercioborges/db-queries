select from_unixtime(l.timecreated,'%Y-%m-%d %H:%i')as hora, count(l.timecreated) as hits from mdl_logstore_standard_log l
WHERE l.timecreated >= unix_timestamp(@inicio) and l.timecreated <= unix_timestamp(@fim)
group by hora
order by hora desc
